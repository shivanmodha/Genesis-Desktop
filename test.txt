    function validateInputs(studentId)
    {
        var testField1 = document.getElementById('fldAuthorizedAdultPhone-' + studentId);
        var testField2 = document.getElementById('fldAuthorizedAdult-' + studentId);
        if (testField1.value == "" || testField2.value == "")
            return "You must enter Authorized Adult/Activity and Contact Number, and then select the checkbox.";
        else
            return "";
    }

    function showDailyPickupDialog(studentId, dailyPickupReadOnly, releaseType)
    {
        $( "#dialog-dailyPickup-" + studentId ).dialog({
            resizable: false,
            height:'auto',
            width:580,
            modal: true,
            buttons:
            {
                Save: function()
                {
                    var validationMessage = validateInputs(studentId);

                    if (validationMessage == "")
                    {
                        var acknowledged = $('#fldAcknowledgement-' + studentId).is(':checked');
                        if (acknowledged)
                        {
                            $( this ).dialog( "close" );
                            document['frmDailyPickup-' + studentId].submit();
                        }
                        else
                        {
                            alertDialog('Please indicate that you have finished filling out this form by selecting the checkbox.');
                        }
                    }
                    else
                    {
                        alertDialog(validationMessage);
                    }
                },
                Cancel: function()
                {
                    $( this ).dialog( "close" );
                }
            },
            open: function()
            {
                if (dailyPickupReadOnly)
                    $(".ui-dialog-buttonpane button:contains('Save')").button('disable');

                $('#fldReleaseType-' + studentId).val(releaseType);
                var noteTag = "noteField_" + studentId;
                var text = $('#'+noteTag).text();
                $('#fldNotes-' + studentId).val(text);
            }
        });
    }

    function deleteDailyPickupRelease(workStudentId, navStudentId, readOnly)
    {
        if (readOnly)
        {
            alertDialog('Pickup releases cannot be deleted after 12pm, please call your school to make any changes.');
            return;

        {
            location.href = 'parents?tab1=studentdata&tab2=studentsummary&action=deleteDailyPickupRelease&studentid=' + navStudentId + '&workStudentId=' + workStudentId;
        });

    }

        function confirmDialog(questionToAsk, okFunction, cancelFunction)
        {
            $( "#_dialog-confirm_" ).dialog({
                resizable: false, height:'auto', width:600, modal: true,
                title: 'Confirmation',
                open: function()
                {
                    $('#_confirmDialogText_').html(questionToAsk);
                },
                buttons:
                {
                    Ok: function()
                    {
                        $( this ).dialog( "close" );
                        if (okFunction!=null)
                            okFunction();
                    },
                    Cancel: function()
                    {
                        $( this ).dialog( "close" );
                        if (cancelFunction!=null)
                            cancelFunction();
                    }
                }
            });
        }

        function alertDialog(alertText, okCallback)
        {
            $( "#_dialog-alert_" ).dialog({
                resizable: false, height:'auto', width:600, modal: true,
                title: 'Information',
                open: function()
                {
                    $('#_confirmAlertText_').html(alertText);
                },
                buttons:
                {
                    Ok: function()
                    {
                        $( this ).dialog( "close" );
                        if (okCallback)
                            okCallback();
                    }
                }
            });
        }

        function promptDialog(questionToAsk, defaultAnswer, okFunction, cancelFunction)
        {
            if (defaultAnswer==null) defaultAnswer = '';
            $( "#_dialog-prompt_" ).dialog({
                resizable: false, height:'auto', width:600, modal: true,
                title: 'Prompt',
                open: function()
                {
                    $('#_confirmPromptText_').html(questionToAsk);
                    $('#_prompt_dialog_value_').val(defaultAnswer).removeClass('fieldValueChanged').addClass('fieldValue').focus().select();
                },
                buttons:
                {
                    Ok: function()
                    {
                        $( this ).dialog( "close" );
                        if (okFunction!=null)
                        {
                            var __prompt__answer = $('#_prompt_dialog_value_').val();
                            if (__prompt__answer == null) __prompt__answer = '';
                            $('#_prompt_dialog_value_').removeClass('fieldValueChanged').addClass('fieldValue');
                            okFunction(__prompt__answer);
                        }
                    },
                    Cancel: function()
                    {
                        $( this ).dialog( "close" );
                        if (cancelFunction!=null)
                            cancelFunction();
                    }
                }
            });
        }
